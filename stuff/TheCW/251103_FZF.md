**FullName : FuZzy file Finder**

### 終端下使用

```zsh
# 上下鍵綁定 預覽文件腳本
export FZF_DEFAULT_OPTS='
--bind ctrl-j:down,ctrl-k:up
--preview"[[ $(file --mime {}) =~ binary ]] && echo {} is a binary file || (ccat --color=always {} || highlight -O ansi -l {} || cat {}) 2> /dev/null | head -500"'
 
# 查找選項 使用silver_searcher工具 查找隱藏文件 忽略.git文件
export FZF_DEFAULT_COMMAND='ag --hidden --ignore .git -g ""'

# 終端下喚出fzf的字符
export FZF_COMPLETION_TRIGGER='\'

# (沒講） !!
export FZF_PREVIEW_COMMAND='[[ $(file --mime {}) =~ binary ]] && echo {} is a binary file || (ccat --color=always {} || highlisht -O ansi -l || cat {}) 2> /dev/null | head -500'
#export FZF_TMUX_HEIGHT='80%'

# 來自鏈接!!
source ~/.config/zsh/completion.zsh
# 這個文件在倉庫中已被刪除
#source ~/.config/zsh/key-bindings.zsh
```

### 聯動 vim

```vim
      Plug 'junegunn/fzf', { 'do': { -> fzf#install() } }
      Plug 'junegunn/fzf.vim'
      "...
      
noremap <C-p> :FZF<CR>
noremap <C-f> :Ag<CR>
noremap <C-h> :MRU<CR>
noremap <C-t> :BTags<CR>
noremap <C-l> :LinesWithPreview<CR>
noremap <C-w> :Buffers<CR>
noremap q;    :History:<CR>

autocmd! FileType fzf
autocmd  FileType fzf set laststatus=0 noruler
    \| autocmd BufLeave <buffer> set laststatus=2 ruler

command! -bang -nargs=* Buffers
    \ call fzf#vim#buffers(
    \   '',
    \   <bang>0 ? fzf#vim#with_preview('up:60%')
    \           : fzf#vim#with_preview('right:0%', '?'),
    \   <bang>0)
    
command! -bang -nargs=* LinesWithPreview
    \ call fzf#vim#grep(
    \   'rg --with-filename --column --line-number --no-heading --color=always --smart-case . '.fnameescape(expand('%')), 1,
    \   fzf#vim#with_preview({}, 'up:50%', '?'),
    \   1)
    
command! -bang -nargs=* Ag
    \ call fzf#vim#ag(
    \   '',
    \   <bang>0 ? fzf#vim#with_preview('up:60%')
    \           : fzf#vim#with_preview('right:50%', '?'),
    \   <bang>0)

command! -bang -nargs=* MRU call fzf#vim#history(fzf#vim#with_preview())

command! -bang BTags
    \ call fzf#vim#buffer_tags('', {
    \   'down': '40%',
    
    }
```
